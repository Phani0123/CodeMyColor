Gradient Examples

            

                

                    

                        Text on Gradient
                    

                    

                        
/* Basic Linear Gradient */
.element {
  background: linear-gradient(45deg, #ff3333, #3333ff);
  color: white;
  text-shadow: 0 0 10px rgba(0, 0, 0, 0.5);
}

                    

                

                
                

                    

                        Radial Gradient
                    

                    

                        
/* Radial Gradient */
.element {
  background: radial-gradient(circle, #ff3333, #3333ff);
}

                    

                

                
                

                    

                        Animated Gradient
                    

                    

                        
/* Animated Gradient */
@keyframes gradient-animation {
  0% { background-position: 0% 50%; }
  100% { background-position: 100% 50%; }
}

.element {
  background: linear-gradient(to right, #ff3333, #3333ff);
  background-size: 200% 100%;
  animation: gradient-animation 3s infinite alternate;
}

                    

                

                
                

                    

                        Conic Gradient
                    

                    

                        
/* Conic Gradient */
.element {
  background: conic-gradient(from 0deg, #ff3333, #3333ff, #33ff33, #ff3333);
}

                    

                

                
                

                    

                        Repeating Gradient
                    

                    

                        
/* Repeating Gradient */
.element {
  background: repeating-linear-gradient(
    45deg,
    #ff3333,
    #ff3333 10px,
    #3333ff 10px,
    #3333ff 20px
  );
}

                    

                

                
                

                    

                        Gradient Overlay
                    

                    

                        
/* Gradient Overlay on Image */
.element {
  background: linear-gradient(
    rgba(255, 51, 51, 0.8),
    rgba(51, 51, 255, 0.8)
  ), url('image.jpg');
  background-size: cover;
}
